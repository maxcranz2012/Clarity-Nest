// Пример локальной базы данных const documents = [ { title: "Документ 1", content: "Это содержание документа 1" }, { title: "Документ 2", content: "Это содержание документа 2" }, { title: "Документ 3", content: "Это содержание документа 3" }, // Добавьте больше документов по необходимости ]; function getQueryParam(param) { const urlParams = new URLSearchParams(window.location.search); return urlParams.get(param); } function search() { const query = getQueryParam('query').toLowerCase(); const resultsContainer = document.getElementById('results-container'); resultsContainer.innerHTML = ''; // Очищаем предыдущие результаты if (query) { const results = documents.filter(doc => doc.content.toLowerCase().includes(query) || doc.title.toLowerCase().includes(query)); if (results.length > 0) { results.forEach(doc => { const resultItem = document.createElement('div'); resultItem.innerHTML = `

${doc.title}

${doc.content}

`; resultsContainer.appendChild(resultItem); }); } else { resultsContainer.innerHTML = '

Нет результатов

'; } } } document.addEventListener('DOMContentLoaded', search); if ('serviceWorker' in navigator) { window.addEventListener('load', function() { navigator.serviceWorker.register('service-worker.js') .then(function(registration) { console.log('Service Worker registered with scope:', registration.scope); }).catch(function(error) { console.log('Service Worker registration failed:', error); }); }); }
